新建 `sakura_backend/core/minigames.py`：

```python
"""内置小游戏：猜谜、成语接龙、快问快答"""
import random
import json

class GameBase:
    def __init__(self):
        self.running = False

class GuessGame(GameBase):
    """文字猜谜"""
    def __init__(self):
        super().__init__()
        self.riddles = [
            {"q": "什么东西越洗越脏？", "a": "水"},
            {"q": "什么东西不能吃？", "a": "亏"},
        ]
    
    def start(self) -> str:
        self.running = True
        self.current = random.choice(self.riddles)
        return f"猜谜开始！{self.current['q']}"
    
    def check(self, answer: str) -> str:
        if not self.running:
            return "没有进行中的游戏"
        if answer.strip() == self.current['a']:
            self.running = False
            return "答对了！🎉"
        return "不对哦，再想想~"

class IdiomGame(GameBase):
    """成语接龙"""
    def __init__(self):
        super().__init__()
        self.last_word = ""
    
    def start(self) -> str:
        self.running = True
        self.last_word = "一心一意"
        return f"成语接龙开始！我说：{self.last_word}"
    
    def check(self, word: str) -> str:
        if not self.running:
            return "没有进行中的游戏"
        if word[0] != self.last_word[-1]:
            return f"要以「{self.last_word[-1]}」开头哦"
        self.last_word = word
        return f"接得好！该我了……{word}（你的回合）"

# 游戏注册表
GAMES = {
    "猜谜": GuessGame,
    "成语": IdiomGame,
}

def get_game(name: str):
    cls = GAMES.get(name)
    return cls() if cls else None
```

}
```

"""彩蛋系统：关键词触发特殊回复"""
import random
import datetime

EASTER_EGGS = {
    "我饿了": ["饿了就去吃点好的~", "想吃啥？我给你推荐"],
    "好无聊": ["那我陪你聊聊天吧", "要不要玩个小游戏？(/play)"],
    "晚安": ["晚安，做个好梦~", "晚安！明天见"],
}

# 节日彩蛋
HOLIDAY_EGGS = {
    "01-01": "元旦快乐！新的一年也要开心哦🎉",
    "02-14": "情人节快乐~有情人终成眷属💕",
    "12-25": "圣诞快乐🎄！",
}

def check_easter_egg(text: str) -> str | None:
    """检查是否触发彩蛋"""
    for keyword, replies in EASTER_EGGS.items():
        if keyword in text:
            if random.random() < 0.3:  # 30%概率触发
                return random.choice(replies)
    return None

def check_holiday() -> str | None:
    """检查是否节日"""
    today = datetime.datetime.now().strftime("%m-%d")
    return HOLIDAY_EGGS.get(today)
```

集成到 `sakura_backend/api/chat_session.py` 的 `_handle_chat_message()`：
```python
from core.easter_eggs import check_easter_egg, check_holiday
egg = check_easter_egg(self.user_message) or check_holiday()
if egg:
    # 彩蛋作为首条回复的一部分
    reply = egg + "\n" + reply
```

**涉及文件**：

**验收标准**：

"""快捷命令链：一键触发多步操作"""
quick_replies = {}  # name -> [commands]

def register(name: str, commands: list):
    quick_replies[name] = commands

def execute(name: str) -> str:
    cmds = quick_replies.get(name)
    if not cmds:
        return f"没有找到快捷命令：{name}"
    results = []
    for cmd in cmds:
        results.append(f"[执行] {cmd}")
    return "\n".join(results)

# 预设快捷命令（可通过 API 自定义）
register("早安", ["查天气", "报日程", "说早安"])
register("晚安", ["道晚安", "设明早提醒"])

### 基础实现（MiMo 做）

```python
def detect_chapter_boundary(messages: list) -> bool:
    """检测是否到达章节边界"""
    if len(messages) < 2:
        return False
    
    last = messages[-1]["content"] if messages else ""
    
    # 边界条件：话题切换/日期变更/重大事件
    topic_shifters = ["对了", "话说", "说起来", "突然想到"]
    for shifter in topic_shifters:
        if any(shifter in last for shifter in topic_shifters):
            return True
    
    return False

def generate_chapter_title(episodes: list) -> str:
    """为章节生成标题"""
    if not episodes:
        return "新章节"
    # 用第一个 episode 的摘要作为标题基础
    topic = episodes[0].get("topic_summary", "")
    date = episodes[0].get("start_time", "")[:10]
    if topic:
        return f"{date} {topic[:20]}"
    return f"第{date}章"
```

class FlashbackPipe(PromptPipe):
    """记忆闪回：当话题和过去相似时触发"""
    zone = "dynamic"
    def process(self, ctx):
        if ctx.msg_counter % 50 != 0:  # 每50条最多1次
            return None
        if not ctx.user_message:
            return None
        
        from core.vector_store import search_similar
        results = search_similar(ctx.user_message, top_k=3)
        if not results:
            return None
        
        # 检查是否有高度相似的过去对话
        doc, dist = results[0]
        if dist < 0.3:  # 非常相似
            return f"（夕语突然想起……上次你也是这样说的：「{doc[:30]}」）\n"
        return None
```

**涉及文件**：

**验收标准**：
- 用户重复类似话题时触发闪回
- 每50条最多触发1次，不频繁

"""故事线追踪：跟踪跨对话的故事弧"""
from core.db import get_conn
import json
from datetime import datetime

def ensure_arc_table():
    with get_conn() as conn:
        conn.cursor().execute("""CREATE TABLE IF NOT EXISTS story_arcs (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT, status TEXT DEFAULT 'active',
            progress TEXT DEFAULT '{}',
            created_at TEXT, updated_at TEXT
        )""")

def create_arc(name: str) -> int:
    ensure_arc_table()
    with get_conn() as conn:
        cursor = conn.cursor()
        cursor.execute("INSERT INTO story_arcs (name, created_at, updated_at) VALUES (?, ?, ?)",
                      (name, datetime.now().isoformat(), datetime.now().isoformat()))
        return cursor.lastrowid

def get_active_arcs() -> list:
    ensure_arc_table()
    with get_conn() as conn:
        cursor = conn.cursor()
        cursor.execute("SELECT * FROM story_arcs WHERE status='active' ORDER BY updated_at DESC")
        return [dict(r) for r in cursor.fetchall()]
```

```python
def generate_recall_report(days: int = 7) -> str:
    """生成回忆报告"""
    from datetime import datetime, timedelta
    cutoff = (datetime.now() - timedelta(days=days)).isoformat()
    
    with get_conn() as conn:
        cursor = conn.cursor()
        cursor.execute(
            "SELECT topic_summary, emotion_summary, importance, start_time FROM episodes WHERE start_time >= ? ORDER BY start_time DESC",
            (cutoff,)
        )
        episodes = cursor.fetchall()
    
    if not episodes:
        return f"近{days}天没有聊天记录"
    
    total_episodes = len(episodes)
    top_episode = max(episodes, key=lambda e: e["importance"])
    
    return f"""📊 近{days}天聊天报告
━━━━━━━━━━━━━
聊天次数：{total_episodes} 次
最重要的话题：「{top_episode['topic_summary']}」
情感基调：{' '.join(e['emotion_summary'] for e in episodes if e['emotion_summary'])}"""
```

**涉及文件**：
- `sakura_backend/core/episodic_memory.py`（~25行）

**验收标准**：
- 生成可读的回忆报告
- 内容包含次数、重要话题、情感
